\chapter{Team Ranking --- Solutions}

\begin{exercise}{5.1}
Four teams with given results. (a) Find rankings. (b) Predict T1 vs T4.
\end{exercise}
\begin{solution}
Games: T1--T2(1), T1--T3(1), T2--T3(2), T2--T4(1), T3--T4(1). Point totals: $q_1=-1$, $q_2=13$, $q_3=-8$, $q_4=-4$ (check: sum $=0$ $\checkmark$).
\[M=\begin{pmatrix}2&-1&-1&0\\-1&4&-2&-1\\-1&-2&4&-1\\1&1&1&1\end{pmatrix},\quad\bar{q}=\begin{pmatrix}-1\\13\\-8\\0\end{pmatrix}.\]
\textbf{Rankings:} $r_1=0.125$, $r_2=2.375$, $r_3=-1.125$, $r_4=-1.375$. Order: T2$>$T1$>$T3$>$T4.

\textbf{(b):} Expected margin $=r_1-r_4=1.5$. T1 expected to win by 1.5.
\end{solution}

\begin{exercise}{5.2}
Four teams. (a) Find rankings. (b) Predict T2 vs T4.
\end{exercise}
\begin{solution}
Same $M$ as 5.1 (same game counts). $\bar{q}=(38,-21,-23,0)^T$.
\textbf{Rankings:} $r_1=13.5$, $r_2=-5.333$, $r_3=-5.667$, $r_4=-2.5$. Order: T1$>$T4$>$T2$\approx$T3.

\textbf{(b):} $r_2-r_4=-2.833$. T4 expected to win by about 2.8.
\end{solution}

\begin{exercise}{5.3}
Read off structural info from a given Massey equation.
\end{exercise}
\begin{solution}
$M=\begin{pmatrix}3&-1&-2&0\\-1&4&-2&-1\\-2&-2&4&0\\1&1&1&1\end{pmatrix}$.

\textbf{(a)} 4 teams. \textbf{(b)} T4 original diagonal entry: $|m_{42}|=1$, so 1 game (vs T2). \textbf{(c)} T1: $m_{11}=3$ games; $|m_{12}|=1$ vs T2, $|m_{13}|=2$ vs T3 (twice). \textbf{(d)} Original $m_{44}=1$. Total $=(3+4+4+1)/2=6$ games.
\end{solution}

\begin{exercise}{5.4}
Show the sum of diagonal entries of $M$ (before modification) is always even.
\end{exercise}
\begin{solution}
$\sum_i m_{ii}=\sum_i(\text{games team }i\text{ played})=2G$ (each game is counted in exactly 2 teams' totals). Since $2G$ is even, done. $\square$
\end{solution}

\begin{exercise}{5.5}
Four teams with home advantage. (a) Without home, (b) reduce home wins by 25\%.
\end{exercise}
\begin{solution}
\textbf{(a):} $q_1=15,q_2=-22,q_3=14,q_4=-7$. Rankings: $r_1=6.5,r_2=-4.0,r_3=2.0,r_4=-4.5$.

\textbf{(b):} Reduce home wins: T1 at home beats T2 by $5\to3.75$; T3 at home beats T4 by $21\to15.75$.
$q_1=3.75+10=13.75$, $q_2=-3.75-10+7-14=-20.75$, $q_3=-10-7+15.75=-1.25$, $q_4=14-15.75=-1.75$.
Rankings (b): $r_1\approx4.125,r_2\approx-4.375,r_3\approx-1.125,r_4\approx1.375$. T4 moved above T3.
\end{solution}

\begin{exercise}{5.6}
If T1 beats T2 by $k$, T2 beats T3 by $k$, T3 beats T1 by $k$: show equal rankings.
\end{exercise}
\begin{solution}
$q_1=k-k=0$, $q_2=-k+k=0$, $q_3=-k+k=0$. With $\bar{q}=\mathbf{0}$ and the constraint row, the unique solution is $\hat{\vr}=\mathbf{0}$. All teams have equal ranking 0. $\square$
\end{solution}

\begin{exercise}{5.9 / 5.10}
Better to beat team twice by 1 or once by 2?
\end{exercise}
\begin{solution}
\textbf{(a) Once by 2:} T1-T2(1), T2-T3(1), T1-T3(1) with margins 3, 2, 2.
$M=\begin{pmatrix}2&-1&-1\\-1&2&-1\\1&1&1\end{pmatrix}$, $\bar{q}=(5,-1,0)^T$. Rankings: $r_1\approx1.667$.

\textbf{(b) Twice by 1:} T1-T3 now twice (total margin 2).
$M=\begin{pmatrix}3&-1&-2\\-1&2&-1\\1&1&1\end{pmatrix}$, $\bar{q}=(5,-1,0)^T$. Rankings: $r_1\approx1.067$.

One win by 2 gives T1 a higher ranking than two wins by 1 each.
\end{solution}

\begin{exercise}{5.11}
(a) T1 beats T2 by $n$, T2 beats T3 by 1, T3 beats T1 by 1. Find $\lim_{n\to\infty}\hat{\vr}_n$. (b) T1 beats T2 by 1 $n$ times instead.
\end{exercise}
\begin{solution}
\textbf{(a):} $q_1=n-1$, $q_2=-(n-1)$, $q_3=0$. By symmetry $r_3=0$, $r_1=-r_2$. Row 1: $3r_1=n-1$, so $r_1=(n-1)/3\to\infty$.

\textbf{(b):} Now $m_{11}=n+1$, $m_{12}=m_{21}=-n$. Same point totals but $(2n+1)r_1=n-1$, so $r_1=(n-1)/(2n+1)\to 1/2$.

Winning once by $n$ is unlimited; winning $n$ times by 1 converges to a bounded limit.
\end{solution}

\begin{exercise}{5.12}
T1 beats T2 by $\alpha$, T2 beats T3 by $\beta$, T3 beats T1 by $\gamma$. When are all rankings equal?
\end{exercise}
\begin{solution}
$q_1=\alpha-\gamma$, $q_2=-\alpha+\beta$, $q_3=-\beta+\gamma$. All zero $\Leftrightarrow \alpha=\beta=\gamma$.

Two teams can tie (e.g., $\alpha=2,\beta=4,\gamma=0$ gives $r_1=r_3$). Spread is unbounded as $\alpha\to\infty$.
\end{solution}

\begin{exercise}{5.13}
Two groups $\{T1,T2,T3\}$ and $\{T4,T5,T6\}$ with no inter-group games. (a) Massey fails; (b) rank groups; (c) after T3 beats T4 by 1.
\end{exercise}
\begin{solution}
\textbf{(a):} Two disconnected components $\Rightarrow$ block-singular Massey matrix.

\textbf{(b):} Each group has $M_{\text{sub}}=\begin{pmatrix}1&-1&0\\-1&2&-1\\1&1&1\end{pmatrix}$, $\bar{q}=(2,1,0)^T$.
Rankings: $r_1\approx2.333,r_2\approx0.333,r_3\approx-2.667$ (same for T4,T5,T6).
\begin{lstlisting}
M6=[1 -1 0 0 0 0;-1 2 -1 0 0 0;0 -1 2 -1 0 0;
    0 0 -1 2 -1 0;0 0 0 -1 2 -1;1 1 1 1 1 1];
q6=[2;-2+3;-3+1;-1+2;-2+3;0]; r6=M6\q6;
\end{lstlisting}
\textbf{(c) Output:} $r_1\approx5.33,r_2\approx3.33,r_3\approx0.33,r_4\approx-0.67,r_5\approx-2.67,r_6\approx-5.67$.
\end{solution}

\begin{exercise}{5.16}
Nontrivial examples where no LS is needed; is sum constraint still required?
\end{exercise}
\begin{solution}
\textbf{Example:} T1 beats T2 by 3, T2 beats T3 by 3, T1 beats T3 by 6. The three equations $r_1-r_2=3$, $r_2-r_3=3$, $r_1-r_3=6$ are consistent, so the original $A\vr=\vp$ has an exact solution.

\textbf{Sum constraint still needed:} Even with consistent results, the Massey matrix $A^TA$ is singular (its rows sum to 0). Without $\sum r_i=0$, there are infinitely many solutions differing by a constant shift. The constraint gives the unique centered solution.
\end{solution}
