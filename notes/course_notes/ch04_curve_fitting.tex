\section{Least-Squares Line}
%% ================================================================

Given points $(x_1,y_1),\ldots,(x_n,y_n)$, fit $f(x)=mx+b$. Writing one equation per point:
\[
  \underbrace{\begin{pmatrix}x_1&1\\\vdots&\vdots\\x_n&1\end{pmatrix}}_{A}
  \begin{pmatrix}m\\b\end{pmatrix} = \begin{pmatrix}y_1\\\vdots\\y_n\end{pmatrix} = \mathbf{b}.
\]

Solve the normal equation $A^TA\hat{\mathbf{c}}=A^T\mathbf{b}$. This minimizes $\sum(y_i-f(x_i))^2$ (sum of squared \emph{vertical} distances).

\begin{remark}
The line $y=mx+b$ and the line $x=ny+c$ (which minimizes horizontal distances) are generally \textbf{different} because they minimize different objective functions.
\end{remark}

\section{General Function Fitting: What Works?}
%% ================================================================

Least squares applies whenever the unknown coefficients appear \textbf{linearly}:
\[
  f(x) = a_1 f_1(x) + \cdots + a_k f_k(x),
\]
where $f_1,\ldots,f_k$ are known functions. The matrix is:
\[
  A = \begin{pmatrix}f_1(x_1)&\cdots&f_k(x_1)\\\vdots&&\vdots\\f_1(x_n)&\cdots&f_k(x_n)\end{pmatrix}.
\]

\begin{example}[What Works and What Doesn't]
\begin{itemize}
  \item $f(x)=ae^x+bx$: \textbf{Yes} (linear in $a,b$). Matrix column $i$: $(e^{x_i},x_i)$.
  \item $f(x)=a\sin x+b\cos x+c$: \textbf{Yes} (linear in $a,b,c$).
  \item $f(x)=e^{ax}+bx$: \textbf{No} ($a$ appears in exponent).
  \item $f(x)=a\sin(bx)+c$: \textbf{No} ($b$ appears inside sine).
  \item $f(x)=ax^2+bx^c$: \textbf{No} ($c$ appears as exponent).
\end{itemize}
\end{example}

\section{Common Fitting Templates}
%% ================================================================

\subsection*{Polynomial Fit}

\textbf{Parabola} $f(x)=ax^2+bx+c$: column $i$ of $A$ is $(x_i^2,x_i,1)$.

\textbf{Degree-$k$ polynomial}: column $i$ of $A$ is $(x_i^k,x_i^{k-1},\ldots,x_i,1)$.

\subsection*{Exponential Fit}

$f(x)=ae^x+b$: column $i$ of $A$ is $(e^{x_i},1)$.

\subsection*{Sinusoidal Fit}

$f(x)=a+b\sin x$: column $i$ of $A$ is $(1,\sin x_i)$.

\subsection*{Surface Fitting}

For $f(x,y)=a_1g_1(x,y)+\cdots+a_kg_k(x,y)$, each data point $(x_i,y_i,z_i)$ gives one row. Example: paraboloid $f(x,y)=ax^2+by^2$, matrix row $i$: $(x_i^2,y_i^2)$.

\subsection*{Ellipse Fitting (Polar Form)}

For an ellipse $Ar^2\cos^2\theta + Br^2\sin^2\theta = 1$, rewrite as:
\[
  (r_i\cos\theta_i)^2 \cdot A + (r_i\sin\theta_i)^2 \cdot B = 1.
\]
Matrix row $i$: $(r_i^2\cos^2\theta_i,\; r_i^2\sin^2\theta_i)$, right-hand side all 1s.

\section{Worked Examples}
%% ================================================================

\begin{example}[Line Through Three Points]
Points $(-1,0),(1,2),(2,2)$.
\[
A=\begin{pmatrix}-1&1\\1&1\\2&1\end{pmatrix}, \quad \mathbf{b}=\begin{pmatrix}0\\2\\2\end{pmatrix}.
\]
$A^TA=\begin{pmatrix}6&2\\2&3\end{pmatrix}$, $A^T\mathbf{b}=\begin{pmatrix}5\\4\end{pmatrix}$, $\det=14$.
\[
\hat{\mathbf{c}}=\frac{1}{14}\begin{pmatrix}3&-2\\-2&6\end{pmatrix}\begin{pmatrix}5\\4\end{pmatrix}=\begin{pmatrix}1/2\\11/7\end{pmatrix}.
\]
\textbf{Line:} $f(x)=\tfrac{1}{2}x+\tfrac{11}{7}\approx0.5x+1.571$.
\end{example}

\begin{example}[Parabola Fit]
Points $(-1,3),(0,1),(1,2),(3,9)$. Set up $A$ with columns $(x_i^2,x_i,1)$ and solve. The least-squares parabola is approximately $f(x)\approx 1.045x^2-0.555x+1.291$.
\end{example}

\begin{example}[Sinusoidal Fit]
Points $(-2,6.3),(3,1.2),(5,7.1),(8,-2.8),(9,-0.05)$. Fit $f(x)=a+b\sin x$.
\[
A=\begin{pmatrix}1&\sin(-2)\\1&\sin3\\1&\sin5\\1&\sin8\\1&\sin9\end{pmatrix}.
\]
Solving: $f(x)\approx 2.026-4.974\sin x$. Predict $f(\pi/2)\approx-2.948$.
\end{example}

\section{Using the Best-Fit Function}
%% ================================================================

After fitting:
\begin{enumerate}
  \item Evaluate $f$ at any point to make predictions.
  \item Solve $f(x)=c$ to find when the function reaches a target.
  \item Compute $f'$ to find rates of change.
  \item Compare the least-squares errors of competing models to select the best fit.
\end{enumerate}
